bash <<'DEPLOY_EOF'
set -Eeuo pipefail

REPO_URL="https://github.com/yddfhbh/kannyan.git"
BRANCH="main"
APP_DIR="$HOME/discord-bot-new"
OLD_APP_DIR="$HOME/discord-bot"
DATA_DIR="$HOME/discord-bot-data"
BACKUP_DIR="$HOME/discord-bot-backups"
ENV_BACKUP="$HOME/.discord-bot.env"

trap 'echo; echo "[ERROR] 배포 실패: line $LINENO"; echo "마지막 상태 확인:"; cd "$APP_DIR" 2>/dev/null && git status --short || true; pm2 status || true' ERR

log() {
  echo
  echo "========== $* =========="
}

backup_state_files() {
  local src="$1"
  local dst="$2"

  [ -d "$src" ] || return 0
  mkdir -p "$dst"

  find "$src" -maxdepth 1 -type f \
    \( -name '*.json' -o -name '*.sqlite' -o -name '*.db' -o -name '.env' \) \
    -exec cp -a {} "$dst"/ \; 2>/dev/null || true
}

copy_old_data_without_overwrite() {
  local src="$1"

  if [ -d "$src" ] && [ ! -L "$src" ]; then
    find "$src" -mindepth 1 -maxdepth 1 ! -name daily-chess-puzzle.json -exec cp -an {} "$DATA_DIR"/ \; 2>/dev/null || true
  fi
}

remove_repo_data_before_git() {
  cd "$APP_DIR"

  if [ -L data ]; then
    echo "repo 안의 data 심볼릭 링크만 제거합니다: $(readlink data)"
    rm -f data
  elif [ -e data ]; then
    local data_backup="$BACKUP_DIR/data-from-repo-before-git-$TS"
    echo "repo 안에 일반 data 경로가 있어서 백업으로 이동합니다: $data_backup"
    mv data "$data_backup"
  fi
}

ensure_lichess_db() {
  mkdir -p data

  if [ -s data/lichess_db_puzzle.csv.zst ] && zstd -t data/lichess_db_puzzle.csv.zst >/dev/null 2>&1; then
    echo "기존 lichess_db_puzzle.csv.zst 사용."
    return 0
  fi

  echo "Lichess puzzle DB 다운로드 중..."
  rm -f data/lichess_db_puzzle.csv.zst.tmp
  curl -fL --retry 3 --retry-delay 5 \
    https://database.lichess.org/lichess_db_puzzle.csv.zst \
    -o data/lichess_db_puzzle.csv.zst.tmp

  zstd -t data/lichess_db_puzzle.csv.zst.tmp >/dev/null
  mv data/lichess_db_puzzle.csv.zst.tmp data/lichess_db_puzzle.csv.zst
}

log "[1/12] Backing up .env"
mkdir -p "$BACKUP_DIR"

if [ -r "$APP_DIR/.env" ]; then
  cp "$APP_DIR/.env" "$ENV_BACKUP"
elif [ -r "$OLD_APP_DIR/.env" ]; then
  cp "$OLD_APP_DIR/.env" "$ENV_BACKUP"
fi

log "[2/12] Stopping old bot"
pm2 delete discord-bot || true
pkill -f "[d]iscord-bot/src/index.js" || true
pkill -f "[d]iscord-bot-new/src/index.js" || true

log "[3/12] Preparing persistent data dir and state backups"
mkdir -p "$DATA_DIR"
mkdir -p "$BACKUP_DIR"

TS="$(date +%Y%m%d-%H%M%S)"

backup_state_files "$DATA_DIR" "$BACKUP_DIR/state-persistent-$TS"
backup_state_files "$APP_DIR/data" "$BACKUP_DIR/state-discord-bot-new-$TS"
backup_state_files "$OLD_APP_DIR/data" "$BACKUP_DIR/state-discord-bot-$TS"

log "[4/12] Merging daily chess state"
node --input-type=module - <<'NODE_EOF'
import fs from 'node:fs/promises';

const paths = [
  '/home/ubuntu/discord-bot-data/daily-chess-puzzle.json',
  '/home/ubuntu/discord-bot-new/data/daily-chess-puzzle.json',
  '/home/ubuntu/discord-bot/data/daily-chess-puzzle.json',
];

async function readJson(path) {
  try {
    return JSON.parse(await fs.readFile(path, 'utf8'));
  } catch {
    return {};
  }
}

function normalizeState(raw) {
  const state = raw && typeof raw === 'object' ? raw : {};

  state.settings ??= {};
  state.settings.guilds ??= {};
  state.posts ??= {};
  state.solved ??= {};
  state.sessions ??= {};

  for (const [key, value] of Object.entries(state)) {
    if (
      /^\d+$/.test(key) &&
      value &&
      typeof value === 'object' &&
      typeof value.channelId === 'string'
    ) {
      state.settings.guilds[key] ??= value;
    }
  }

  return state;
}

function mergeState(base, nextRaw) {
  const next = normalizeState(nextRaw);

  return {
    ...base,
    ...next,
    settings: {
      ...(base.settings ?? {}),
      ...(next.settings ?? {}),
      guilds: {
        ...(base.settings?.guilds ?? {}),
        ...(next.settings?.guilds ?? {}),
      },
    },
    posts: {
      ...(base.posts ?? {}),
      ...(next.posts ?? {}),
    },
    solved: {
      ...(base.solved ?? {}),
      ...(next.solved ?? {}),
    },
    sessions: {
      ...(base.sessions ?? {}),
      ...(next.sessions ?? {}),
    },
  };
}

let merged = normalizeState({});

for (const path of paths) {
  merged = mergeState(merged, await readJson(path));
}

await fs.mkdir('/home/ubuntu/discord-bot-data', { recursive: true });

if (
  Object.keys(merged.settings.guilds).length > 0 ||
  Object.keys(merged.posts).length > 0 ||
  Object.keys(merged.solved).length > 0 ||
  Object.keys(merged.sessions).length > 0
) {
  await fs.writeFile(
    '/home/ubuntu/discord-bot-data/daily-chess-puzzle.json',
    JSON.stringify(merged, null, 2),
    'utf8'
  );

  console.log('Merged daily-chess-puzzle.json');
  console.log(JSON.stringify(merged.settings.guilds, null, 2));
} else {
  console.log('No existing daily chess state to merge.');
}
NODE_EOF

log "[5/12] Copying old data files without overwriting persistent data"
copy_old_data_without_overwrite "$APP_DIR/data"
copy_old_data_without_overwrite "$OLD_APP_DIR/data"

log "[6/12] Installing system packages"
sudo apt update
sudo apt install -y \
  git \
  curl \
  zstd \
  fontconfig \
  fonts-dejavu-core \
  fonts-noto-core \
  fonts-noto-extra \
  fonts-noto-cjk \
  fonts-noto-color-emoji

sudo fc-cache -f -v >/dev/null || true

log "[7/12] Syncing repo with Git"
if [ ! -d "$APP_DIR/.git" ]; then
  echo "Git repo가 없어서 새로 clone합니다."
  rm -rf "$APP_DIR"
  git clone --branch "$BRANCH" "$REPO_URL" "$APP_DIR"
else
  remove_repo_data_before_git
fi

cd "$APP_DIR"
git remote set-url origin "$REPO_URL"

git fetch --prune origin "$BRANCH"
git checkout -B "$BRANCH" "origin/$BRANCH"
git reset --hard "origin/$BRANCH"
git clean -fd -e .env

echo "Current commit:"
git log -1 --oneline

log "[8/12] Linking persistent data dir"
rm -rf data
ln -sfn "$DATA_DIR" data

echo "data link:"
ls -ld data
echo "real data path:"
readlink -f data

if [ ! -f scripts/build-lichess-puzzle-pool.js ]; then
  echo "scripts/build-lichess-puzzle-pool.js가 없습니다. Git 상태를 확인하세요."
  exit 1
fi

if grep -R -n --include='*.js' --include='*.mjs' --include='*.cjs' "Replies with Pong\|commandName === 'ping'\|commandName === \"ping\"" src; then
  echo "예전 ping 코드가 남아 있어서 배포를 중단합니다."
  exit 1
fi

log "[9/12] Restoring .env"
if [ -r "$ENV_BACKUP" ]; then
  cp "$ENV_BACKUP" .env
fi

if [ ! -f .env ]; then
  if [ ! -r /dev/tty ]; then
    echo ".env가 없고 현재 세션에서 입력을 받을 수 없습니다."
    echo "먼저 $APP_DIR/.env 또는 $ENV_BACKUP 을 만들어주세요."
    exit 1
  fi

  echo ".env가 없어서 지금 새로 만듭니다."
  read -r -s -p "DISCORD_TOKEN 입력: " DISCORD_TOKEN_VALUE </dev/tty
  echo
  cat > .env <<ENV_FILE_EOF
DISCORD_TOKEN=$DISCORD_TOKEN_VALUE
CLIENT_ID=1502588698246778960

ENV_FILE_EOF
fi

if ! grep -E '^(GEMINI_API_KEYS|GEMMA_API_KEYS|GEMINI_API_KEY|GEMMA_API_KEY)=' .env | cut -d= -f2- | grep -q '[^[:space:]]'; then
  if [ ! -r /dev/tty ]; then
    echo "Gemma/Gemini API 키가 .env에 없고 현재 세션에서 입력을 받을 수 없습니다."
    echo ".env에 GEMINI_API_KEYS=... 를 추가한 뒤 다시 실행하세요."
    exit 1
  fi

  echo "Gemma/Gemini API 키가 .env에 없어서 추가합니다."
  read -r -s -p "GEMINI_API_KEYS 입력: " GEMINI_API_KEYS_VALUE </dev/tty
  echo
  printf '\nGEMINI_API_KEYS=%s\n' "$GEMINI_API_KEYS_VALUE" >> .env
fi

if ! grep -q '^GEMMA_MODEL=' .env && ! grep -q '^GEMINI_MODEL=' .env; then
  echo "GEMMA_MODEL=gemma-4-26b-a4b-it" >> .env
fi

if ! grep -q '^GEMMA_FALLBACK_MODELS=' .env && ! grep -q '^GEMINI_FALLBACK_MODELS=' .env; then
  echo "GEMMA_FALLBACK_MODELS=gemma-4-31b-it" >> .env
fi

if ! grep -q '^DAILY_CHESS_PUZZLE_HOUR=' .env; then
  echo "DAILY_CHESS_PUZZLE_HOUR=0" >> .env
fi

cp .env "$ENV_BACKUP"

log "[10/12] Installing node dependencies"
if [ -f package-lock.json ]; then
  npm ci --omit=dev
else
  npm install --omit=dev
fi

log "[11/12] Checking/building Lichess puzzle pools"
if [ ! -s data/lichess-puzzle-pool.jsonl ]; then
  echo "lichess-puzzle-pool.jsonl 없음. 새로 생성합니다."
  ensure_lichess_db

  LICHESS_PUZZLE_MIN_RATING=2600 \
  LICHESS_PUZZLE_MAX_RATING=3400 \
  LICHESS_PUZZLE_MAX_RD=120 \
  LICHESS_PUZZLE_MIN_POPULARITY=60 \
  LICHESS_PUZZLE_MIN_PLAYS=30 \
  zstd -dc data/lichess_db_puzzle.csv.zst | node scripts/build-lichess-puzzle-pool.js
else
  echo "기존 lichess-puzzle-pool.jsonl 사용."
fi

if [ -f scripts/build-lichess-puzzle-rush-pool.js ]; then
  if [ ! -s data/lichess-puzzle-rush-pool.jsonl ]; then
    echo "lichess-puzzle-rush-pool.jsonl 없음. 새로 생성합니다."
    ensure_lichess_db
    zstd -dc data/lichess_db_puzzle.csv.zst | node scripts/build-lichess-puzzle-rush-pool.js
  else
    echo "기존 lichess-puzzle-rush-pool.jsonl 사용."
  fi
else
  echo "scripts/build-lichess-puzzle-rush-pool.js 없음. 퍼즐러쉬 풀 생성은 건너뜁니다."
fi

echo "Puzzle pool line counts:"
wc -l data/lichess-puzzle-pool.jsonl 2>/dev/null || true
wc -l data/lichess-puzzle-rush-pool.jsonl 2>/dev/null || true

log "[12/12] Registering commands and starting bot"
npm run register

for f in src/daily-chess-puzzle.js src/index.js; do
  if [ -f "$f" ]; then
    node --check "$f"
  fi
done

if ! command -v pm2 >/dev/null 2>&1; then
  sudo npm install -g pm2 || npm install -g pm2
fi

pm2 delete discord-bot || true

if [ -f ecosystem.config.cjs ]; then
  pm2 start ecosystem.config.cjs --update-env
else
  pm2 start src/index.js --name discord-bot --update-env
fi

pm2 save
pm2 status

sleep 3

echo
echo "Health:"
curl -s http://127.0.0.1:8080/health || true
echo

echo
echo "Daily chess state:"
node --input-type=module - <<'NODE_EOF'
import fs from 'node:fs/promises';

try {
  const state = JSON.parse(await fs.readFile('data/daily-chess-puzzle.json', 'utf8'));
  console.log('guild settings:');
  console.log(JSON.stringify(state.settings?.guilds ?? {}, null, 2));
  console.log('posts keys:', Object.keys(state.posts ?? {}));
  console.log('sessions keys:', Object.keys(state.sessions ?? {}));
} catch (error) {
  if (error.code === 'ENOENT') {
    console.log('daily-chess-puzzle.json 없음. /일일퍼즐지정 실행하면 생성됩니다.');
  } else {
    throw error;
  }
}
NODE_EOF

echo
echo "Persistent data dir:"
echo "$DATA_DIR"
ls -lah "$DATA_DIR"

echo
echo "PM2 logs:"
pm2 logs discord-bot --lines 80 --nostream || true

echo
echo "배포 완료."
DEPLOY_EOF
